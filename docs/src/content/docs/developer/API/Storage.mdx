```ts
import { JarvisStore } from 'jarvis-api/ui';

const store = new Store('setting.bin');
const store = new Store(); // filename is optional. Set a filename if you plan to have multiple store files. e.g. Separate settings and data.

await store.set('key', 'value');
await store.save();

const value = await store.get('key');
console.log(value);
```
